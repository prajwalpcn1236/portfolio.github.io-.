Implementation Code 

//BFS
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

class Graph {
    int V; // number of vertices
    vector<vector<int>> adj;

public:
    Graph(int V) : V(V), adj(V) {}

    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u); // for undirected graph
    }

    void BFS(int start) {
        vector<bool> visited(V, false);
        queue<int> q;

        visited[start] = true;
        q.push(start);

        while (!q.empty()) {
            int u = q.front(); q.pop();
            cout << u << " ";

            for (int v : adj[u]) {
                if (!visited[v]) {
                    visited[v] = true;
                    q.push(v);
                }
            }
        }
    }
};

int main() {
    Graph g(5);
    g.addEdge(0,1);
    g.addEdge(0,2);
    g.addEdge(1,3);
    g.addEdge(2,4);

    cout << "BFS traversal starting from node 0: ";
    g.BFS(0);
    cout << endl;

    return 0;
}

‚õ∞Ô∏è 2. Heaps (Priority Queues)

‚úÖ Why Used:

To prioritize pricing updates based on urgency and market impact.

Efficiently handles real-time price adjustment requests and competitive pricing alerts.

‚è±Ô∏è Time Complexity:

Insert: O(log n)

Extract min/max: O(log n)

Peek: O(1)

üß† Space Complexity:

O(n) where n is the number of elements

Binary heap implementation using arrays

Implementation Code 

#include <iostream>
#include <queue>
#include <vector>

using namespace std;

int main() {
    // Min-heap using priority_queue with greater comparator
    priority_queue<int, vector<int>, greater<int>> minHeap;

    minHeap.push(30);
    minHeap.push(10);
    minHeap.push(20);
    minHeap.push(5);

    cout << "Elements in min-heap order:\n";
    while (!minHeap.empty()) {
        cout << minHeap.top() << " ";
        minHeap.pop();
    }
    cout << endl;

    return 0;
}
